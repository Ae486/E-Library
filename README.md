```bash
# 1. 克隆你的仓库
git clone https://github.com/Ae486/E-Librarian.git
cd E-Librarian

# 2. 双击 start.bat（Windows）
# 或者手动执行：

# 安装 Python 依赖
cd backend
pip install -r requirements.txt

# 安装 Node.js 依赖
cd ../frontend
npm install

# 构建 React 应用
npm run build

# 启动服务器
cd ../backend
python app.py
```

### 前置要求
- Python 3.7+
- Node.js 18+
- TypeScript
- Vite

```bash
npm install -g typescript
npm install -g vite
```
